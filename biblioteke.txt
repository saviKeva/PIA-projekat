//brzi upiti za mongo bazu
db.reservations.find({
    od: {
        $gte: new Date("2026-03-02T00:00:00Z"),
        $lte: new Date("2026-03-08T23:59:59Z")
    }
}).forEach(function(doc) {
    let stariOd = new Date(doc.od);
    let stariDo = new Date(doc.do);

    // Pomera datum za 7 dana
    stariOd.setDate(stariOd.getDate() + 7);
    stariDo.setDate(stariDo.getDate() + 7);

    db.reservations.updateOne(
        { _id: doc._id },
        {
            $set: {
                od: stariOd,
                do: stariDo,
                status: "kreirana"
            }
        }
    );
});
